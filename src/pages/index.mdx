export const GuestList = () => {
  const guests = useGuests()
  return (
    <>
      {guests
        .filter(guest => !!guest.nickname)
        .map((guest, idx) => {
          return (
            <React.Fragment key={guest.slug}>
              <ol>
                <Link to={`/introduce-yourself/${guest.slug}`}>
                  {`${idx + 1}. ${guest.title}`}
                </Link>
              </ol>
            </React.Fragment>
          )
        })}
    </>
  )
}

<GuestList />
